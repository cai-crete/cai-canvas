### Change A+B: pen/eraser 터치 → 임시 pan

`~line 960–982` (pen/eraser 모드 touch 차단 부분):

```ts
// 변경 전
} else if (canvasMode === 'pen') {
  if (e.pointerType === 'touch') return; // V267

} else if (canvasMode === 'eraser') {
  if (e.pointerType === 'touch') return; // V267

// 변경 후
} else if (canvasMode === 'pen') {
  if (e.pointerType === 'touch') {
    // V286 A: 손가락 터치 → 임시 pan (tempRestoreCanvasModeRef 활용)
    tempRestoreCanvasModeRef.current = 'pen';
    setCanvasMode('pan');
    isDraggingPanRef.current = true;
    setIsDraggingPan(true);
    dragStartRef.current = { x: e.clientX - canvasOffset.x, y: e.clientY - canvasOffset.y };
    e.currentTarget.setPointerCapture(e.pointerId);
    return;
  }

} else if (canvasMode === 'eraser') {
  if (e.pointerType === 'touch') {
    // V286 B: 손가락 터치 → 임시 pan (tempRestoreCanvasModeRef 활용)
    tempRestoreCanvasModeRef.current = 'eraser';
    setCanvasMode('pan');
    isDraggingPanRef.current = true;
    setIsDraggingPan(true);
    dragStartRef.current = { x: e.clientX - canvasOffset.x, y: e.clientY - canvasOffset.y };
    e.currentTarget.setPointerCapture(e.pointerId);
    return;
  }
```

## 동작 흐름

```
pen/eraser 활성 상태
  └─ 손가락 pointerDown
       → tempRestoreCanvasModeRef = 'pen'|'eraser'
       → setCanvasMode('pan')
       → isDraggingPanRef = true (팬 시작)
  └─ 손가락 pointerMove → 캔버스 팬 이동 ✓
  └─ 손가락 pointerUp
       → isDraggingPanRef = false
       → tempRestoreCanvasModeRef 있음 → setCanvasMode('pen'|'eraser') ← 자동 복귀 ✓

스타일러스 pointerDown (pen/eraser 모드)
  → pointerType === 'pen' → touch 분기 안 탐 → 드로잉 정상 동작 ✓
```
